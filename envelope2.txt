import matplotlib.pyplot as plt
import numpy as np
from __future__ import division

t_max=2 #改変箇所をまとめておく
x_max=7
y_max=6
y_min=-5
step=1/3

t_min=-t_min #対称性の利用
x_min=-x_max

def f(x,t):
	return t*x-t**2

def subplots():
	fig,ax=plt.subplots()
	
	for spine in ['left','bottom']:
		ax.spines[spine].set_position('zero')
	for spine in ['right','top']:
		ax.spines[spine].set_color('none')
	return fig,ax

fig,ax=subplots()
x=np.linspace(x_min,x_max,200)

#tが整数値以外も取れるように、初項がt_min、項差がstepの等差数列とみなした
n_max=2*t_max/step

for n in arange(0,n_max+1,1):
	t=t_min+n*step
	y=f(x,t)
	ax.plot(x,y,color='black',linewidth=1)

for tick in ax.yaxis.get_major_ticks(): #y軸の目盛削除
	tick.tick1On=False
	tick.label1On=False

for tick in ax.xaxis.get_major_ticks(): #x軸の目盛削除
	tick.tick1On=False
	tick.label1On=False

xlim(x_min,x_max)
ylim(y_min,y_max)
plt.show()

#上端と右端の変な点を消したい。。。
